#Variables ‘OECD’, ‘Europe_CentralAsia’, ‘LatinAmer_Caribb’, ‘MidEast_NAfrica’, ‘EastAsia_Pacific’, ‘SouthAsia’, ‘SubSahaAfrica’ created with corresponding countries
country_suicides <- SuicideRates1985_2016 %>% select("country", "year", "suicides/100k pop")
# Vector variable called ‘countrysuicides2014’ created to extract the year 2014 from the data
countrysuicides2014 <- c("2014")
country_suicides <- country_suicides [ grepl(paste(countrysuicides2014, collapse="|"), country_suicides$year),]
# Calculating the average suicides of in each country in 2014
suicidesper100k <- as.numeric(country_suicides$`suicides/100k pop`)
country_suicides <- cbind(country_suicides, suicidesper100k)
country_suicides <- country_suicides %>% select(-"suicides/100k pop")
country_suicides <- country_suicides %>% group_by(country) %>% summarise("Avg. Suicide per 100k people" = mean(suicidesper100k))
world_suicide_rates <- left_join(map.world, country_suicides, by = c('region' = 'country'))
ggplot(data = world_suicide_rates, aes(x = long, y = lat, group = group)) +
geom_polygon(aes(fill = `Avg. Suicide per 100k people`)) + 
scale_fill_viridis_c(option = 'magma') + 
labs(title = "Suicide Rates Around the World (2014)", caption = "Suicide rates per 100k people") + theme_minimal()
